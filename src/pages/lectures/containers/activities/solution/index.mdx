This is the solution to challenge [Docker lab](../)

1. docker -v
2. docker run hello-world
3. docker images
4. docker build -t <image_name>:<image_tag> .
5. docker images
6. docker run -p 9080:9080 -d --name greeting greeting:v1.0.0
7. docker ps
8. docker inspect greeting
9. docker logs greeting
10. docker login quay.io
11. docker tag greeting:v1.0.0 quay.io/<repository_name>/greeting:v1.0.0
12. docker push quay.io/<repository_name>/greeting:v1.0.0
13. docker stop greeting
14. docker rm greeting
15. docker rmi <image_id>
16. docker pull quay.io/<repository_name>/greeting:v1.0.0
17. docker build -f Dockerfile.multistage -t greeting:v2.0.0 .
18. docker run -p 9080:9080 -d --name greeting_multistage greeting:v2.0.0
19. docker ps
20. docker inspect greeting_multistage
21. docker logs greeting_multistage
22. docker stop greeting_multistage
23. docker rm greeting_multistage 
24. docker rmi <image_id>
25. docker images or docker images | grep greeting
